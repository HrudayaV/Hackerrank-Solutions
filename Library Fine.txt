def libraryFine(d1, m1, y1, d2, m2, y2):
    '''
    In this problem going back to past to submit the book is also possible.
    '''
    if y1 - y2 > 0:
        return 10000
    elif y1 == y2 and m1 - m2 > 0:
        return (m1 - m2) * 500
    elif y1 == y2 and m1 == m2 and d1 - d2 > 0:
        return (d1 - d2) * 15
    else:
        return 0
